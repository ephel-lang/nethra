enum Nat:
  case Zero()
  case Succ[P <: Nat]()

type Add[P1 <: Nat, P2 <: Nat] <: Nat =
  P1 match
    case Zero => P2
    case Succ[p] => Succ[Add[p, P2]]

sealed trait Vec[S <: Nat, E]:
  def +[L <: Nat](l: Vec[L, E]): Vec[Add[S, L], E]

case class Empty[E]() extends Vec[Zero, E]:
  override def +[L <: Nat](l: Vec[L, E]) = l

case class Cons[S <: Nat, E](head: E, tail: Vec[S, E])
  extends Vec[Succ[S], E]:
  override def +[L <: Nat](l: Vec[L, E]) = Cons(head, tail.+(l))
