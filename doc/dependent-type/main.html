    val zero : Nat = fold (inl unit)
    val succ : Nat -> Nat = fun n -> fold (inr n)

    sig add : Nat -> Nat -> Nat
    val add = fun n1 n2 ->
        case (unfold n1)
             (fun _  -> n2)
             (fun n1 -> succ (add n1 n2))
