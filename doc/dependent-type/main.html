Basic examples with Nethra


    sig Nat : type
    val Nat = rec(X:type).(Unit | X)

    val zero : Nat = fold (inl unit)
    val succ : Nat -> Nat = fun n -> fold (inr n)

    sig add : Nat -> Nat -> Nat
    val add = fun n1 n2 ->
        case (unfold n1)
             (fun _  -> n2)
             (fun n1 -> succ (add n1 n2))
            

    val Monoid : type -> type = fun t -> sig struct
        sig neutral : t
        sig combine : t -> t -> t
        sig law1    : {a:t} -> a = (combine neutral a)
    end

    val MonoidNat : Monoid Nat = val struct
        val neutral = zero
        val combine = add
        val law1    = refl
    end
            

    val Monoid : type = sig struct
        sig t       : type
        sig neutral : t
        sig combine : t -> t -> t
        sig law1    : {a:t} -> a = (combine neutral a)
    end

    val MonoidNat : Monoid = val struct
        val t       = Nat
        val neutral = zero
        val combine = add
        val law1    = refl
    end
            

    val Functor : (type -> type) -> type = fun M ->
        sig struct
            sig map : {a b:type} -> (a -> b) -> M a -> M b
            -{ Functor laws specification }-
            sig `map id = id` : {a:type} ->
                let lhd = map (id {a}) in
                let rhd = id {M a} in
                lhd = rhd
            sig `map (f o g) = (map f) o (map g)` :
                {a b c:type} -> (f:b -> c) -> (g:a -> b) ->
                let lhd = map (compose f g) in
                let rhd = compose (map f) (map g) in
                lhd = rhd
        end
            

    sig transitivity :
        {A:type} -> {a b c :A}
        -> a = b
        -> b = c
        --------
        -> a = c

    val transitivity = fun a=b b=c ->
        subst (subst refl by a=b) by b=c
            

    sig congruent_app :
        {A B:type} -> (f g:A -> B)
        -> f = g
        ------------------------------
        -> {a:A} -> (f a) = (g a)

    val congruent_app = fun f g f=g ->
        subst refl by (f=g)
